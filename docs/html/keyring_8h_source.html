146 rapidjson::Document readFromKeyring() {
147 rapidjson::Document d;
148 std::unique_ptr<GError> err = nullptr;
149 GError* errPtr = err.get();
150
151 const gchar* result = secret_password_lookupv_sync(
152 &m_schema, m_attributes.getGHashTable(), nullptr, &errPtr);
153
154 if (err) {
155 throw std::runtime_error(err->message);
156 }
157
158 if (result != nullptr && strcmp(result, "") != 0 &&
159 !d.Parse(result).HasParseError()) {
160 if (strcmp(result, "null") == 0) {
161 const char* json = "{}";
162 d.Parse(json);
163 }
164 return d;
165 }
166
167 this->storeToKeyring(d);
