28inline void LoggingPrologue() {

31#if defined(ENABLE_DLT)

36 switch (msg.level) {
37 case SPDLOG_LEVEL_TRACE:
38 Dlt::LogString(DltLogLevelType::LOG_VERBOSE, msg.payload.data());
39 break;
40 case SPDLOG_LEVEL_DEBUG:
41 Dlt::LogString(DltLogLevelType::LOG_DEBUG, msg.payload.data());
42 break;
43 case SPDLOG_LEVEL_INFO:
44 Dlt::LogString(DltLogLevelType::LOG_INFO, msg.payload.data());
45 break;
46 case SPDLOG_LEVEL_WARN:
47 Dlt::LogString(DltLogLevelType::LOG_WARN, msg.payload.data());
48 break;
49 case SPDLOG_LEVEL_ERROR:
50 Dlt::LogString(DltLogLevelType::LOG_ERROR, msg.payload.data());
51 break;
52 case SPDLOG_LEVEL_CRITICAL:
53 Dlt::LogString(DltLogLevelType::LOG_FATAL, msg.payload.data());
